-( x - 7) - ( 2x - 4) = 5
-x + 7 - 2x + 4 = 5
-3x = -6
x = 2 tidak memenuhi karena yg di minta < dari 2

Untuk 2_< x < 7
-( x - 7) + ( 2x - 4 ) = 5
-x + 7 + 2x - 4 = 5
x = 2 memenuhi karena 2_< 2 < 7

Untuk x _> 7
( x - 7 ) + ( 2x - 4 ) = 5
x - 7 + 2x - 4 = 5
3x = 16
x = 16/3
