Untuk bilangan real positif a, b, c, p, q, dan r berlaku:
[tex]\begin{aligned}\frac{a}{p}=\frac{b}{q}=\frac{c}{r}\end{aligned}[/tex]

Ditanyakan
[tex]\begin{aligned}\frac{abc\cdot(p+q)(q+r)(r+p)}{pqr\cdot(a+b)(b+c)(c+a)}={\dots}\end{aligned}[/tex]

PENYELESAIAN

[tex]\begin{aligned}&\frac{a}{p}=\frac{b}{q}=\frac{c}{r}\\&{\implies}\begin{cases}(1)&aq=bp\\(2)&br=cq\\(3)&ar=cp\end{cases}\end{aligned}[/tex]

Kemudian, kita selesaikan.

[tex]\begin{aligned}&\frac{abc\cdot(p+q)(q+r)(r+p)}{pqr\cdot(a+b)(b+c)(c+a)}\\{=\ }&\frac{a(p+q)\cdot b(q+r)\cdot c(r+p)}{p(a+b)\cdot q(b+c)\cdot r(c+a)}\\{=\ }&\frac{(ap+aq)}{(ap+bp)}\cdot\frac{(bq+br)}{(bq+cq)}\cdot\frac{(cr+cp)}{(cr+ar)}\end{aligned}[/tex]
[tex]\begin{aligned}{=\ }&\frac{\overbrace{(ap+bp)}^{\!\!\!\begin{array}{c}(1):bp\to aq\end{array}\!\!\!}}{(ap+bp)}\ \cdot\ \frac{\overbrace{(bq+cq)}^{\!\!\!\begin{array}{c}(2):cq\to br\end{array}\!\!\!}}{(bq+cq)}\ \cdot\ \frac{\overbrace{(cr+ar)}^{\!\!\!\begin{array}{c}(3):ar\to cp\end{array}\!\!\!}}{(cr+ar)}\\{=\ }&1\cdot1\cdot1\\{=\ }&\boxed{\:\bf1\:}\end{aligned}[/tex]

KESIMPULAN

[tex]\begin{aligned}\therefore\ \frac{abc\cdot(p+q)(q+r)(r+p)}{pqr\cdot(a+b)(b+c)(c+a)}=\boxed{\:\bf1\:}\end{aligned}[/tex]
[tex]\blacksquare[/tex]
