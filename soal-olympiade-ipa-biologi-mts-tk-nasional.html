39. MATHEMATIC OLYMPIADE SD


segitiga 1 =
[tex] \frac{a \times t}{2} = \frac{7 \times 3}{2} = 10 \frac{1}{2} [/tex]
segitiga 2 =
[tex] \frac{a \times t}{2} = \frac{7 \times 6}{2} = 21 - 10 \frac{1}{2 } = 10 \frac{1}{2} [/tex]
segitiga 3 =
[tex] \frac{a \times t}{2} = \frac{7 \times 9}{2} = 31 \frac{1}{2} - 10 \frac{1}{2} = 21[/tex]
maka segitiga 2 + segitiga 3
[tex] 10\frac{1}{2} + 21 = 31 \frac{1}{2} [/tex]
jawabannya adalah C
